from enum import Enum

from typing import List, Optional, Union

class ComparisonType(str, Enum):

class Aggressiveness(str, Enum):

-- messages table
CREATE TABLE setup_messages (
  id        SERIAL PRIMARY KEY,
  setup_date DATE      NOT NULL,
  raw_text   TEXT      NOT NULL,
  created_at TIMESTAMP DEFAULT NOW()
);

-- one row per ticker in each message
CREATE TABLE ticker_setups (
  id               SERIAL PRIMARY KEY,
  message_id       INTEGER  REFERENCES setup_messages(id),
  symbol           TEXT     NOT NULL
);

-- signals for each ticker
CREATE TABLE signals (
  id               SERIAL PRIMARY KEY,
  ticker_setup_id  INTEGER   REFERENCES ticker_setups(id),
  category         TEXT      NOT NULL,
  aggressiveness   TEXT      NOT NULL,
  comparison       TEXT      NOT NULL,
  trigger_price    DOUBLE PRECISION,
  trigger_low      DOUBLE PRECISION,
  trigger_high     DOUBLE PRECISION
);

-- target levels per signal
CREATE TABLE signal_targets (
  id        SERIAL PRIMARY KEY,
  signal_id INTEGER REFERENCES signals(id),
  level     DOUBLE PRECISION NOT NULL,
  position  INTEGER          NOT NULL  -- 1,2,3 for target ordering
);

-- bias rules per ticker
CREATE TABLE biases (
  id               SERIAL PRIMARY KEY,
  ticker_setup_id  INTEGER   REFERENCES ticker_setups(id),
  direction        TEXT      NOT NULL,
  condition        TEXT      NOT NULL,
  price            DOUBLE PRECISION NOT NULL,
  flip_comparison  TEXT,
  flip_price       DOUBLE PRECISION
);
How to Use
Ingestion:

Create a TradeSetupMessage, store it in setup_messages.

For each TickerSetup, insert into ticker_setups and parse its Signal and Bias records.

Querying:

Bias rules live in their own table for easy lookups when evaluating exit conditions.
